function doPost(e) { try { var data = JSON.parse(e.postData.contents); var action = data.action; var ss = SpreadsheetApp.getActiveSpreadsheet(); // ========================================== // AUTO-BANKING (ĐÓN WEBHOOK TỪ SMSFORWARDER) // ========================================== // SmsForwarder thường bọc nội dung thông báo vào trường "content" hoặc "message" var rawText = data.content || data.message || data.sms_content || data.text; if (rawText) { rawText = String(rawText).toUpperCase(); // 1. Dùng Regex tóm Số Tiền (Tìm dấu + và các con số/dấu phẩy/dấu chấm đi liền sau nó) // Ví dụ nó sẽ tóm được: +10,000 hoặc +50.000 hoặc +100000 var amountMatch = rawText.match(/\+([\d,\.]+)/); // 2. Dùng Regex tóm 6 số cuối SĐT ngụy trang // NÂNG CẤP: Chỉ tóm đúng cụm nào có CHÍNH XÁC 6 SỐ (bỏ qua các mã dài ngoằng của MoMo/Bank) var phoneMatch = rawText.match(/(?:^|\D)(\d{6})(?:\D|$)/); // Nếu tìm thấy CẢ số tiền CẢ 6 số điện thoại thì mới làm tiếp if (amountMatch && phoneMatch) { // Dọn dẹp số tiền (cắt bỏ dấu phẩy, dấu chấm) để thành số chuẩn toán học var amount = parseInt(amountMatch[1].replace(/[,.]/g, '')); var phoneToCharge = phoneMatch[1]; var sheetUsers = ss.getSheetByName("Users"); var dataUsers = sheetUsers.getDataRange().getValues(); for (var i = 1; i < dataUsers.length; i++) { var dbPhone = String(dataUsers[i][0]).trim(); if (dbPhone.endsWith(phoneToCharge)) { var currentBalance = parseInt(dataUsers[i][4]) || 0; var newBalance = currentBalance + amount; sheetUsers.getRange(i + 1, 5).setValue(newBalance); // Cộng tiền // --- GHI LỊCH SỬ GIAO DỊCH (NẠP TIỀN) --- var sheetTrans = ss.getSheetByName("Transactions"); if (sheetTrans) { // Kiểm tra xem có tab Transactions không var transId = "GD" + new Date().getTime(); var timeNow = new Date().toLocaleString("vi-VN"); var noteTrans = "Nạp tự động từ MB Bank"; // Cột: Trans_ID, User_ID (SĐT), Type, Amount, Note, Created_At sheetTrans.appendRow([transId, dbPhone, "Nạp tiền", amount, noteTrans, timeNow]); } // ---------------------------------------- return ContentService.createTextOutput(JSON.stringify({ "success": true, "message": "Nạp tiền SmsForwarder thành công" })).setMimeType(ContentService.MimeType.JSON); } } } return ContentService.createTextOutput("OK").setMimeType(ContentService.MimeType.JSON); } var sheetUsers = ss.getSheetByName("Users"); // ========================================== // 1. ĐĂNG KÝ TÀI KHOẢN (THÊM USERNAME) // ========================================== if (action === "register") { var phone = data.phone; var username = data.username; // Nhận username từ Web var password = data.password; var createdAt = new Date().toLocaleString("vi-VN"); var dataUsers = sheetUsers.getDataRange().getValues(); for (var i = 1; i < dataUsers.length; i++) { // Kiểm tra trùng SĐT if (dataUsers[i][0] == phone) { return ContentService.createTextOutput(JSON.stringify({ "success": false, "message": "Số điện thoại này đã được đăng ký!" })).setMimeType(ContentService.MimeType.JSON); } // Kiểm tra trùng Tên đăng nhập if (dataUsers[i][1] == username) { return ContentService.createTextOutput(JSON.stringify({ "success": false, "message": "Tên đăng nhập này đã có người sử dụng!" })).setMimeType(ContentService.MimeType.JSON); } } // Ghi theo thứ tự: SĐT, Username, Pass, Status, Balance, Time sheetUsers.appendRow([phone, username, password, "Chưa kích hoạt", 0, createdAt]); return ContentService.createTextOutput(JSON.stringify({ "success": true, "message": "Đăng ký thành công! Hãy nhắn Zalo Admin để kích hoạt tài khoản." })).setMimeType(ContentService.MimeType.JSON); } // ========================================== // 2. ĐĂNG NHẬP (TRẢ VỀ CẢ USERNAME) // ========================================== else if (action === "login") { var phone = data.phone; var password = data.password; var dataUsers = sheetUsers.getDataRange().getValues(); for (var i = 1; i < dataUsers.length; i++) { if (dataUsers[i][0] == phone) { if (dataUsers[i][2] == password) { // Mật khẩu giờ ở cột C (index 2) if (dataUsers[i][3] === "Đã kích hoạt") { return ContentService.createTextOutput(JSON.stringify({ "success": true, "username": dataUsers[i][1], // Trả về Tên đăng nhập để hiện lên Web "balance": dataUsers[i][4], // Số dư ở cột E (index 4) "message": "Đăng nhập thành công!" })).setMimeType(ContentService.MimeType.JSON); } else { return ContentService.createTextOutput(JSON.stringify({ "success": false, "message": "Tài khoản đang chờ Admin duyệt!" })).setMimeType(ContentService.MimeType.JSON); } } else { return ContentService.createTextOutput(JSON.stringify({ "success": false, "message": "Sai mật khẩu!" })).setMimeType(ContentService.MimeType.JSON); } } } return ContentService.createTextOutput(JSON.stringify({ "success": false, "message": "Tài khoản không tồn tại!" })).setMimeType(ContentService.MimeType.JSON); } // ========================================== // 3. TẠO ĐƠN HÀNG (TRỪ TIỀN & GHI LOG) // ========================================== else if (action === "create_order") { var note = data.note; var extraNote = data.extraNote; var receiverName = data.receiverName; var phone = data.phone; var totalAmount = parseInt(data.totalAmount) || 0; // Số tiền cần trừ (Web phải gửi lên) var sheetUsers = ss.getSheetByName("Users"); var dataUsers = sheetUsers.getDataRange().getValues(); var userRow = -1; var currentBalance = 0; // 1. Kiểm tra số dư của khách for (var i = 1; i < dataUsers.length; i++) { if (dataUsers[i][0] == phone) { userRow = i + 1; currentBalance = parseInt(dataUsers[i][4]) || 0; // Số dư ở cột E break; } } if (userRow === -1) { return ContentService.createTextOutput(JSON.stringify({"success": false, "message": "Không tìm thấy tài khoản!"})).setMimeType(ContentService.MimeType.JSON); } if (currentBalance < totalAmount) { return ContentService.createTextOutput(JSON.stringify({"success": false, "message": "Số dư không đủ. Vui lòng nạp thêm tiền!"})).setMimeType(ContentService.MimeType.JSON); } // 2. Trừ tiền tài khoản var newBalance = currentBalance - totalAmount; sheetUsers.getRange(userRow, 5).setValue(newBalance); var createdAt = new Date().toLocaleString("vi-VN"); var orderId = "DH" + new Date().getTime(); // 3. Ghi vào bảng Đơn hàng (Orders) var sheetOrders = ss.getSheetByName("Orders"); var productLink = data.productLink; var quantity = data.quantity; var note = data.note; var receiverName = data.receiverName; var receiverPhone = data.receiverPhone; var receiverAddress = data.receiverAddress; var status = "Chờ xử lý"; sheetOrders.appendRow([orderId, phone, productLink, quantity, note, extraNote, receiverName, receiverPhone, receiverAddress, status, createdAt]); // 4. GHI LỊCH SỬ GIAO DỊCH (TRỪ TIỀN MUA HÀNG) var sheetTrans = ss.getSheetByName("Transactions"); if (sheetTrans && totalAmount > 0) { var transId = "GD" + new Date().getTime(); var noteTrans = "Thanh toán mã đơn: " + orderId; sheetTrans.appendRow([transId, phone, "Trừ tiền mua hàng", -totalAmount, noteTrans, createdAt]); // Số tiền âm } return ContentService.createTextOutput(JSON.stringify({ "success": true, "message": "Đã tạo đơn thành công!", "newBalance": newBalance // Trả về số dư mới để Web tự update ngay })).setMimeType(ContentService.MimeType.JSON); } else if (action === "get_orders") { var sheetOrders = ss.getSheetByName("Orders"); var dO = sheetOrders.getDataRange().getValues(); var list = []; for (var i = 1; i < dO.length; i++) { if (dO[i][1] == data.phone) { // Cột I là status (index 8), Cột J là date (index 9) list.push({ orderId: dO[i][0], link: dO[i][2], qty: dO[i][3], note: dO[i][4], status: dO[i][8], date: dO[i][9], mvd: dO[i][10] }); } } return ContentService.createTextOutput(JSON.stringify({"success": true, "orders": list.reverse()})).setMimeType(ContentService.MimeType.JSON); } // HÀM LẤY SỐ DƯ MỚI NHẤT else if (action === "get_balance") { var sheetUsers = ss.getSheetByName("Users"); var dataUsers = sheetUsers.getDataRange().getValues(); for (var i = 1; i < dataUsers.length; i++) { if (dataUsers[i][0] == data.phone) { return ContentService.createTextOutput(JSON.stringify({ "success": true, "balance": dataUsers[i][4] // Số dư đang nằm ở cột E (index 4) })).setMimeType(ContentService.MimeType.JSON); } } } } catch(error) { return ContentService.createTextOutput(JSON.stringify({"success": false, "error": error.message})).setMimeType(ContentService.MimeType.JSON); } }
